async function breed(father, mother) {
  for (let i = 0; i < father.synapses.length; i++) {
    //Randomly select who's genes get used.
    let dna = Math.round(Math.random());
    if (dna === 1) {
      father.synapses[i].weight = mother.synapses[i].weight;
    }
  }
  return father;
}
